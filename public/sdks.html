Quick Install
Node.js
Python
Go
Terminal 
npm install staark-node
JavaScript 
import { Staark } from 'staark-node';

const client = new Staark({
  apiKey: process.env.STAARK_API_KEY,
});

const projects = await client.projects.list();
console.log(projects);
Terminal 
pip install staark-python
Python 
from staark import Staark
import asyncio

client = Staark(api_key="sk_live_staark_••••")

async def main():
    projects = await client.projects.list()
    print(projects)

asyncio.run(main())

go get github.com/staark-app/staark-go
Go 
import "github.com/staark-app/staark-go"

client := staark.New("sk_live_staark_••••")
projects, err := client.Projects.List(ctx)
if err != nil { log.Fatal(err) }
fmt.Println(projects)
